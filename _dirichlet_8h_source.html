2 /*!
3  \file src/DiffEq/Dirichlet.h
4  \copyright 2012-2015, J. Bakosi, 2016-2018, Los Alamos National Security, LLC.
5  \brief Dirichlet SDE
6  \details This file implements the time integration of a system of stochastic
7  differential equations (SDEs), whose invariant is the [Dirichlet
8  distribution](http://en.wikipedia.org/wiki/Dirichlet_distribution).
9 
10  In a nutshell, the equation integrated governs a set of scalars,
11  \f$0\!\le\!Y_\alpha\f$, \f$\alpha\!=\!1,\dots,N-1\f$,
12  \f$\sum_{\alpha=1}^{N-1}Y_\alpha\!\le\!1\f$, as
13  \f[
14  \mathrm{d}Y_\alpha(t) = \frac{b_\alpha}{2}\big[S_\alpha Y_N -
15  (1-S_\alpha)Y_\alpha\big]\mathrm{d}t + \sqrt{\kappa_\alpha Y_\alpha
16  Y_N}\mathrm{d}W_\alpha(t), \qquad \alpha=1,\dots,N-1
17  \f]
18  with parameter vectors \f$b_\alpha > 0\f$, \f$\kappa_\alpha > 0\f$, and \f$0
19  < S_\alpha < 1\f$, and \f$Y_N = 1-\sum_{\alpha=1}^{N-1}Y_\alpha\f$. Here
20  \f$\mathrm{d}W_\alpha(t)\f$ is an isotropic vector-valued [Wiener
21  process](http://en.wikipedia.org/wiki/Wiener_process) with independent
22  increments. The invariant distribution is the Dirichlet distribution,
23  provided the parameters of the drift and diffusion terms satisfy
24  \f[
25  (1-S_1) b_1 / \kappa_1 = \dots = (1-S_{N-1}) b_{N-1} / \kappa_{N-1}.
26  \f]
27  To keep the invariant distribution Dirichlet, the above constraint on the
28  coefficients must be satisfied. For more details on the Dirichlet SDE, see
29  https://doi.org/10.1155/2013/842981.
30 */

93  //! \brief Advance particles according to the Dirichlet SDE
94  //! \param[in,out] particles Array of particle properties
95  //! \param[in] stream Thread (or more precisely stream) ID
96  //! \param[in] dt Time step size
97  void advance( tk::Particles& particles,
98  int stream,
99  tk::real dt,
100  tk::real,
101  const std::map< tk::ctr::Product, tk::real >& )
102  {
103  const auto npar = particles.nunk();
104  for (auto p=decltype(npar){0}; p<npar; ++p) {
105  // Compute Nth scalar
106  tk::real yn = 1.0 - particles(p, 0, m_offset);
107  for (ncomp_t i=1; i<m_ncomp; ++i)
108  yn -= particles( p, i, m_offset );
109 
110  // Generate Gaussian random numbers with zero mean and unit variance
111  std::vector< tk::real > dW( m_ncomp );
112  m_rng.gaussian( stream, m_ncomp, dW.data() );
113 
114  // Advance first m_ncomp (K=N-1) scalars
115  for (ncomp_t i=0; i<m_ncomp; ++i) {
116  tk::real& par = particles( p, i, m_offset );
117  tk::real d = m_k[i] * par * yn * dt;
118  d = (d > 0.0 ? std::sqrt(d) : 0.0);
119  par += 0.5*m_b[i]*( m_S[i]*yn - (1.0-m_S[i]) * par )*dt + d*dW[i];
120  }
121  }
122  }
