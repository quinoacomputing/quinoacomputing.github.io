2 /*!
3  \file src/DiffEq/Beta.h
4  \copyright 2012-2015, J. Bakosi, 2016-2018, Los Alamos National Security, LLC.
5  \brief System of beta SDEs
6  \details This file implements the time integration of a system of stochastic
7  differential equations (SDEs) with linear drift and quadratic diagonal
8  diffusion, whose invariant is the joint [beta
9  distribution](http://en.wikipedia.org/wiki/Beta_distribution).
10 
11  In a nutshell, the equation integrated governs a set of scalars,
12  \f$0\!\le\!Y_\alpha\f$, \f$\alpha\!=\!1,\dots,N\f$, as
13  \f[
14  \mathrm{d}Y_\alpha(t) = \frac{b_\alpha}{2}\left(S_\alpha - Y_\alpha\right)
15  \mathrm{d}t + \sqrt{\kappa_\alpha Y_\alpha(1-Y_\alpha)}
16  \mathrm{d}W_\alpha(t), \qquad \alpha=1,\dots,N
17  \f]
18  with parameter vectors \f$b_\alpha > 0\f$, \f$\kappa_\alpha > 0\f$, and \f$0
19  < S_\alpha < 1\f$. Here
20  \f$\mathrm{d}W_\alpha(t)\f$ is an isotropic vector-valued [Wiener
21  process](http://en.wikipedia.org/wiki/Wiener_process) with independent
22  increments. The invariant distribution is the joint beta distribution. This
23  system of SDEs consists of N independent equations. For
24  more on the beta SDE, see https://doi.org/10.1080/14685248.2010.510843.
25 */

88  //! \brief Advance particles according to the system of beta SDEs
89  //! \param[in,out] particles Array of particle properties
90  //! \param[in] stream Thread (or more precisely stream) ID
91  //! \param[in] dt Time step size
92  void advance( tk::Particles& particles,
93  int stream,
94  tk::real dt,
95  tk::real,
96  const std::map< tk::ctr::Product, tk::real >& )
97  {
98  const auto npar = particles.nunk();
99  for (auto p=decltype(npar){0}; p<npar; ++p) {
100  // Generate Gaussian random numbers with zero mean and unit variance
101  std::vector< tk::real > dW( m_ncomp );
102  m_rng.gaussian( stream, m_ncomp, dW.data() );
103 
104  // Advance all m_ncomp scalars
105  for (ncomp_t i=0; i<m_ncomp; ++i) {
106  tk::real& par = particles( p, i, m_offset );
107  tk::real d = m_k[i] * par * (1.0 - par) * dt;
108  d = (d > 0.0 ? std::sqrt(d) : 0.0);
109  par += 0.5*m_b[i]*(m_S[i] - par)*dt + d*dW[i];
110  }
111  }
112  }
