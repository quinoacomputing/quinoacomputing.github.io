2 /*!
3  \file src/DiffEq/MixMassFractionBeta.h
4  \copyright 2012-2015, J. Bakosi, 2016-2018, Los Alamos National Security, LLC.
5  \brief System of mix mass-fraction beta SDEs
6  \details This file implements the time integration of a system of stochastic
7  differential equations (SDEs) with linear drift and quadratic diagonal
8  diffusion, whose invariant is the joint [beta
9  distribution](http://en.wikipedia.org/wiki/Beta_distribution). There are two
10  differences compared to the plain beta SDE (see DiffEq/Beta.h):
11 
12  - First, the parameters, b, and kappa are specified via functions that
13  constrain the beta SDE to be consistent with the turbulent mixing process.
14  In particular, the SDE is made consistent with the no-mix and fully mixed
15  limits. See, e.g., MixMassFractionBetaCoeffConst::update().
16 
17  - Second, there two additional random variables computed, the same as also
18  computed by the mass-fraction beta equation, see also
19  DiffEq/MassFractionBeta.h.
20 
21  In a nutshell, the equation integrated governs a set of scalars,
22  \f$0\!\le\!Y_\alpha\f$, \f$\alpha\!=\!1,\dots,N\f$, as
23  \f[
24  \mathrm{d}Y_\alpha(t) = \frac{b_\alpha}{2}\left(S_\alpha - Y_\alpha\right)
25  \mathrm{d}t + \sqrt{\kappa_\alpha Y_\alpha(1-Y_\alpha)}
26  \mathrm{d}W_\alpha(t), \qquad \alpha=1,\dots,N
27  \f]
28  with parameter vectors \f$b_\alpha = \Theta b'_\alpha > 0\f$, \f$
29  \newcommand{\irv}[1]{\langle{#1^2}\rangle} \kappa_\alpha = \kappa' \irv{x} >
30  0\f$, and \f$0 < S_\alpha < 1\f$. This is similar to DiffEq/Beta.h, but the
31  parameters, \f$b\f$ and \f$\kappa\f$ constrained. Here \f$
32  \newcommand{\irv}[1]{\langle{#1^2}\rangle}
33  \newcommand{\irmean}[1]{{\langle{#1}\rangle}} \Theta = 1 - \irv{x} /
34  [ \irmean{Y} (1-\irmean{Y}) ]\f$. The fluctuation about the mean, \f$
35  \newcommand{\irmean}[1]{{\langle{#1}\rangle}} \irmean{Y} \f$, is defined as
36  usual: \f$ \newcommand{\irmean}[1]{{\langle{#1}\rangle}} x = Y - \irmean{Y}
37  \f$, and \f$b'\f$ and \f$ \kappa'\f$ are user-specified constants. Also,
38  \f$\mathrm{d}W_\alpha(t)\f$ is an isotropic vector-valued
39  [Wiener process](http://en.wikipedia.org/wiki/Wiener_process) with
40  independent increments. The invariant distribution is the joint beta
41  distribution. This system of SDEs consists of N independent equations. For
42  more on the beta SDE, see https://doi.org/10.1080/14685248.2010.510843.
43 
44  In addition to integrating the above SDE, there are two additional functions
45  of \f$ Y_\alpha \f$ are computed as
46  \f[ \begin{align}
47  \rho(Y_\alpha) & = \frac{ \rho_{2\alpha} }{ 1 + r_\alpha Y_\alpha } \\
48  V(Y_\alpha) & = \frac{1}{ \rho(Y_\alpha) }
49  \end{align} \f]
50  These equations compute the instantaneous mixture density, \f$ \rho \f$, and
51  instantaneous specific volume, \f$ V_\alpha \f$, for equation \f$ \alpha \f$
52  in the system. These quantities are used in binary mixing of
53  variable-density turbulence between two fluids with constant densities, \f$
54  \rho_1, \f$ and \f$ \rho_2 \f$. The additional parameters, \f$ \rho_2 \f$
55  and \f$ r \f$ are user input parameters and kept constant during
56  integration. Since we compute the above variables, \f$\rho,\f$ and \f$V\f$,
57  and call them mixture density and specific volume, respectively, \f$Y\f$,
58  governed by the beta SDE is a mass fraction.
59 
60  _All of this is unpublished, but will be linked in here once published_.
61 */

97  //! \brief Constructor
98  //! \param[in] c Index specifying which system of mix mass-fraction beta
99  //! SDEs to construct. There can be multiple mixmassfracbeta ... end blocks
100  //! in a control file. This index specifies which mix mass-fraction beta
101  //! SDE system to instantiate. The index corresponds to the order in which
102  //! the mixmassfracbeta ... end blocks are given the control file.

104  m_c( c ),

109  get< tag::mixmassfracbeta >().at(c) / 4 ),

111  offset< tag::mixmassfracbeta >(c) ),
112  m_rng( g_rng.at( tk::ctr::raw( g_inputdeck.get< tag::param,
113  tag::mixmassfracbeta,
114  tag::rng >().at(c) ) ) ),
115  m_bprime(),

123  m_ncomp,

125  tag::mixmassfracbeta,
126  tag::bprime >().at(c),
127  g_inputdeck.get< tag::param,
128  tag::mixmassfracbeta,
129  tag::S >().at(c),
130  g_inputdeck.get< tag::param,
131  tag::mixmassfracbeta,
132  tag::kappaprime >().at(c),

135  tag::rho2 >().at(c),

141  // Populate inverse hydrodynamics time scales extracted from DNS, as
142  // associated by the user.
143  if ( Coefficients::type() ==
144  ctr::CoeffPolicyType::HYDROTIMESCALE_HOMOGENEOUS_DECAY )
145  {
146  // Configure inverse hydrodyanmics time scale from DNS
147  const auto& hts = g_inputdeck.get< tag::param,
148  tag::mixmassfracbeta,
149  tag::hydrotimescales >().at(c);

152  Assert( m_hts.size() == m_ncomp, "Number of inverse hydro time scale "
153  "tables associated does not match the components integrated" );

157  tag::mixmassfracbeta,

161  Assert( m_hp.size() == m_ncomp, "Number of hydro "
162  "production/dissipation tables associated does not match the "
163  "components integrated" );
164  }

167  //! Initalize SDE, prepare for time integration
168  //! \param[in] stream Thread (or more precisely stream) ID
169  //! \param[in,out] particles Array of particle properties
170  void initialize( int stream, tk::Particles& particles ) {

182  //! \brief Advance particles according to the system of mix mass-fraction
183  //! beta SDEs
184  //! \param[in,out] particles Array of particle properties
185  //! \param[in] stream Thread (or more precisely stream) ID
186  //! \param[in] dt Time step size
187  //! \param[in] t Physical time of the simulation
188  //! \param[in] moments Map of statistical moments
189  void advance( tk::Particles& particles,
190  int stream,
191  tk::real dt,
192  tk::real t,
193  const std::map< tk::ctr::Product, tk::real >& moments )
194  {
195  // Update SDE coefficients
196  coeff.update( m_depvar, m_ncomp, moments, m_bprime, m_kprime, m_rho2, m_r,
197  m_hts, m_hp, m_b, m_k, m_S, t );
198  // Advance particles
199  const auto npar = particles.nunk();
200  for (auto p=decltype(npar){0}; p<npar; ++p) {
201  // Generate Gaussian random numbers with zero mean and unit variance
202  std::vector< tk::real > dW( m_ncomp );
203  m_rng.gaussian( stream, m_ncomp, dW.data() );
204  // Advance all m_ncomp scalars
205  for (ncomp_t i=0; i<m_ncomp; ++i) {
206  tk::real& Y = particles( p, i, m_offset );
207  tk::real d = m_k[i] * Y * (1.0 - Y) * dt;
208  d = (d > 0.0 ? std::sqrt(d) : 0.0);
209  Y += 0.5*m_b[i]*(m_S[i] - Y)*dt + d*dW[i];
210  // Compute instantaneous values derived from updated Y
211  derived( particles, p, i );
212  }
213  }
214  }

245  //! \brief Return density for mass fraction
246  //! \details Functional wrapper around the dependent variable of the beta
247  //! SDE. This function returns the instantaneous density, rho,
248  //! based on the mass fraction, Y, and parameters rho2 and r'.
249  //! \param[in] Y Instantaneous value of the mass fraction, Y
250  //! \param[in] i Index specifying which (of multiple) parameters to use
251  //! \return Instantaneous value of the density, rho
252  tk::real rho( tk::real Y, ncomp_t i ) const {
253  return m_rho2[i] / ( 1.0 + m_r[i] * Y );
254  }
255 
256  //! \brief Return specific volume for mass fraction
257  //! \details Functional wrapper around the dependent variable of the beta
258  //! SDE. This function returns the instantaneous specific volume, V,
259  //! based on the mass fraction, Y, and parameters rho2 and r'.
260  //! \param[in] Y Instantaneous value of the mass fraction, Y
261  //! \param[in] i Index specifying which (of multiple) parameters to use
262  //! \return Instantaneous value of the specific volume, V
263  tk::real vol( tk::real Y, ncomp_t i ) const {
264  return ( 1.0 + m_r[i] * Y ) / m_rho2[i];
265  }
266 
267  //! Compute instantaneous values derived from updated Y
268  //! \param[in,out] particles Particle properties array
269  //! \param[in] p Particle index
270  //! \param[in] i Component index
271  void derived( tk::Particles& particles, ncomp_t p, ncomp_t i ) const {
272  tk::real& Y = particles( p, i, m_offset );
273  particles( p, m_ncomp+i, m_offset ) = rho( Y, i );
274  particles( p, m_ncomp*2+i, m_offset ) = vol( Y, i );
275  particles( p, m_ncomp*3+i, m_offset ) = 1.0 - Y;
276  }
